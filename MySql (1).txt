create database UEPB
default character set utf8
default collate utf8_general_ci;

use UEPB;

create table campus(
id int primary key,
nome varchar(50) not null,
rua varchar(50),
bairro varchar(50),
numero int,
cep int not null)default character set utf8;

create table aluno(
matricula int primary key,
nome varchar(50),
cpf varchar(9) not null,
foreign key (id_campus) references campus (id_campus)) default character set utf8;


create table responsavel_tecnico(
matricula int primary key,
nome varchar(50),
foreign key (id_campus) references campus (id_campus),
foreign key (referencia_laboratorio) references laboratorio (referencia_laboratorio)) default character set utf8;


create table laboratorio(
referencia int primary key,
capacidade int,
nome varchar(50),
rua varchar(50),
bloco vachar (50),
data_criacao date,
foreign key (id_campus) references campus (id_campus)) default character set utf8;

create table professor(
matricula int primary key,
nome vachar (50),
foreign key (id_campus) references campus (id_campus)) default character set utf8;
